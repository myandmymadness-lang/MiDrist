scripted_gui = {

	imperial = {
		context_type = player_context
		
		window_name = "imperial_container"
		
		visible = {
            has_xiroxito = yes
		}
		
		effects = {
			imperial_bg_click = {
				set_temp_variable = { show = 1 }
				subtract_from_temp_variable = { show = show_imperial_powers }
				set_variable = { show_imperial_powers = show }
			}
            
			imperial_bg_right_click = {
				add_to_variable = { imperial = 0 }
                set_variable = { imperial_max = 500 }
			}
		}
	}
    
}
